        private void Button_Reg_Click(object sender, RoutedEventArgs e)
        {            
                string login = textBoxLogin.Text.Trim();
                string password = PasswordBox.Password.Trim();
                string password2 = PasswordBox2.Password.Trim();
                if (login.Length < 5)//если данные введены не верно
                {
                    textBoxLogin.ToolTip = "Логин должен содержать не менее 5 символов!";
                    textBoxLogin.BorderBrush = Brushes.Red;
                }
                else if (password.Length < 5)
                {
                    PasswordBox.ToolTip = "Пароль должен содержать не менее 5 символов!";
                    PasswordBox.BorderBrush = Brushes.Red;
                }
                else if (password != password2)
                {
                    PasswordBox2.ToolTip = "Пароль введён не верно!";
                    PasswordBox2.BorderBrush = Brushes.Red;
                }
                else
                {
                    textBoxLogin.ToolTip = "";
                    textBoxLogin.Background = Brushes.Transparent;
                    PasswordBox.ToolTip = "";
                    PasswordBox.Background = Brushes.Transparent;
                    PasswordBox2.ToolTip = "";
                    PasswordBox2.Background = Brushes.Transparent;
                    MessageBox.Show("Регистрация прошла успешно!");
                }
        }

        private void Btn_Voiti1(object sender, RoutedEventArgs e)
        {

        }
